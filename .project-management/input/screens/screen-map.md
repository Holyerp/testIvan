```
│   └── /vendors/:id                    → SCREEN-006: VendorDetailScreen

├── /sales                              (ACCOUNTING|MANAGER|ADMIN)
│   ├── /sales/invoices                 → SCREEN-007: SalesInvoiceListScreen

│   ├── /sales/credit-memos             → SCREEN-009: SalesCreditMemoListScreen
│   └── /sales/advance-invoices         → SCREEN-010: SalesAdvanceInvoiceListScreen
│
├── /purchase                           (ACCOUNTING|MANAGER|ADMIN)
│   ├── /purchase/invoices              → SCREEN-011: PurchaseInvoiceListScreen
│   │   └── /purchase/invoices/:id      → SCREEN-012: PurchaseInvoiceDetailScreen
│   └── /purchase/advance-invoices      → SCREEN-013: PurchaseAdvanceInvoiceListScreen
│
├── /inventory                          (WAREHOUSE|MANAGER|ADMIN)
│   ├── /inventory                      → SCREEN-014: InventoryOverviewScreen
│   └── /inventory/items/:id            → SCREEN-015: ItemDetailScreen
│
├── /analytics                          → SCREEN-016: AnalyticsDashboardScreen  (MANAGER|ADMIN)
│
└── /admin                              (ADMIN only)
    ├── /admin/users                    → SCREEN-017: UserManagementScreen
    ├── /admin/settings                 → SCREEN-018: SettingsScreen
    └── /admin/audit-log                → SCREEN-019: AuditLogScreen
```
